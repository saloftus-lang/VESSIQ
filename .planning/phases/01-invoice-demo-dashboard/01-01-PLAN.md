    ```javascript
    document.addEventListener('DOMContentLoaded', () => {
      // Tile entrance animation
      gsap.from('.kpi-tile', {
        y: 20,
        opacity: 0,
        stagger: 0.08,

      });

      // Counter animations
      document.querySelectorAll('.kpi-number').forEach(el => {
        const target = parseFloat(el.dataset.target);
        const prefix = el.dataset.prefix || '';

        const obj = { val: 0 };
        gsap.to(obj, {
          val: target,
          duration: 1.2,
    ```
