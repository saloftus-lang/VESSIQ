    const PORTFOLIO_TOTAL = 847;

    function renderKPIs() {
      const filtered = getFilteredData();

      // Tile 1: Portfolio total (static context label, not filtered)
      document.getElementById('kpi-audited').textContent =
        PORTFOLIO_TOTAL.toLocaleString('en-US');

      // Tile 2: Overcharges found in current filter (per INV-10, D-03)

    **13. GSAP counter animation (Interaction Contract from UI-SPEC):**
    The GSAP animation fires once on page load to animate KPI numbers from 0 to their initial (unfiltered) values. After animation completes, `renderKPIs()` maintains the values on filter changes.

    ```javascript
    document.addEventListener('DOMContentLoaded', () => {
      // Tile entrance animation
      gsap.from('.kpi-tile', {
        y: 20,
        opacity: 0,
        stagger: 0.08,
        duration: 0.5
      });

      // Compute initial KPI values from unfiltered DATA for animation targets
      const allData = DATA;
      const initOvercharges = allData.filter(r => r.overchargeAmount > 0).length;
      const initRecovery = allData
        .filter(r => r.status === 'Won' || r.status === 'Recovered')
        .reduce((sum, r) => sum + r.overchargeAmount, 0);
      const initWon = allData.filter(r => r.status === 'Won').length;
      const initFiled = allData.filter(r => r.status === 'Filed').length;
      const initWinRate = (initWon + initFiled) > 0
        ? Math.round((initWon / (initWon + initFiled)) * 100) : 0;

      // Animate each KPI tile from 0 to computed target
      const kpis = [
        { el: document.getElementById('kpi-audited'), target: PORTFOLIO_TOTAL, prefix: '', suffix: '' },
        { el: document.getElementById('kpi-overcharges'), target: initOvercharges, prefix: '', suffix: '' },
        { el: document.getElementById('kpi-recovery'), target: Math.round(initRecovery), prefix: '$', suffix: '' },
        { el: document.getElementById('kpi-winrate'), target: initWinRate, prefix: '', suffix: '%' },
      ];

      kpis.forEach(({ el, target, prefix, suffix }) => {
        const obj = { val: 0 };
        gsap.to(obj, {
          val: target,
          duration: 1.2,
          ease: 'power2.out',
          onUpdate: () => {
            const v = Math.round(obj.val);
            el.textContent = prefix + v.toLocaleString('en-US') + suffix;
          }
        });
      });

      // After animation, renderKPIs() takes over for filter updates
      // Initial render (populates table placeholder etc.)
      render();
    });
    ```
