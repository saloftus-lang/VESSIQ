    Update the `getFilteredData()` function to also filter by date range:
    ```javascript
    function getFilteredData() {
      const now = new Date();
      return DATA.filter(row => {
        if (state.selectedCarrier !== 'ALL' && row.carrier !== state.selectedCarrier) return false;
        const rowDate = new Date(row.date);
        if (state.dateRange === '1M') {
          const oneMonthAgo = new Date(now);
          oneMonthAgo.setMonth(oneMonthAgo.getMonth() - 1);
          if (rowDate < oneMonthAgo) return false;
        } else if (state.dateRange === '3M') {
          const threeMonthsAgo = new Date(now);
          threeMonthsAgo.setMonth(threeMonthsAgo.getMonth() - 3);
          if (rowDate < threeMonthsAgo) return false;
        }
        // YTD: show all data (data is within current year range)
        return true;
      });
    }
    ```

    Add filter pill click handlers using event delegation:

    document.getElementById('carrierFilters').addEventListener('click', e => {
      const pill = e.target.closest('.filter-pill');
