1. **escapeHtml helper:**

function escapeHtml(str) {
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

2. **Tasks state machine (per D-12, D-13, D-14, RESEARCH Pattern 3):**
```javascript
const TASKS_KEY = 'fnd-tasks';
const DEFAULT_TASKS = [
  { id: 't1', text: 'Follow up with Heather Brown re: Pasha pilot scope', done: false },
  { id: 't2', text: 'Prep ROI summary for first audit batch', done: false },
  { id: 't3', text: 'Identify 2nd pilot prospect outreach', done: false },
];
let tasks = [];

function initTasks() {
  try {
    const saved = localStorage.getItem(TASKS_KEY);
    tasks = saved ? JSON.parse(saved) : [...DEFAULT_TASKS];
    if (!Array.isArray(tasks)) tasks = [...DEFAULT_TASKS];
  } catch (e) {
    tasks = [...DEFAULT_TASKS];
  }
  renderTasks();
}

function saveTasks() {
  localStorage.setItem(TASKS_KEY, JSON.stringify(tasks));
}

function renderTasks() {
  const ul = document.getElementById('task-list');
  if (tasks.length === 0) {
    ul.innerHTML = '<li class="task-empty">No tasks yet. Add your first task above.</li>';
    return;
  }
  ul.innerHTML = tasks.map(t => `
    <li class="task-item${t.done ? ' done' : ''}" data-id="${t.id}">
      <label class="task-check">
        <input type="checkbox" ${t.done ? 'checked' : ''} aria-label="${t.done ? 'Mark task incomplete' : 'Mark task complete'}">
        <span class="task-text">${escapeHtml(t.text)}</span>
      </label>
      <button class="task-delete" data-id="${t.id}" aria-label="Delete task">&times;</button>
    </li>
  `).join('');
}
```

3. **Task event delegation (per RESEARCH Pattern 3):**
```javascript
document.getElementById('task-list').addEventListener('change', e => {
  if (e.target.type !== 'checkbox') return;
  const id = e.target.closest('[data-id]').dataset.id;
  tasks = tasks.map(t => t.id === id ? { ...t, done: e.target.checked } : t);
  saveTasks();
  renderTasks();
});

document.getElementById('task-list').addEventListener('click', e => {
  const btn = e.target.closest('.task-delete');
  if (!btn) return;
  tasks = tasks.filter(t => t.id !== btn.dataset.id);
  saveTasks();
  renderTasks();
});

document.getElementById('task-add-btn').addEventListener('click', () => {
  const input = document.getElementById('task-input');
  const text = input.value.trim();
  if (!text) return;
  tasks.push({ id: 't' + Date.now(), text, done: false });
  input.value = '';
  saveTasks();
  renderTasks();
});

document.getElementById('task-input').addEventListener('keydown', e => {
  if (e.key === 'Enter') document.getElementById('task-add-btn').click();
});
```

4. **Notes auto-save (per D-15, RESEARCH Pattern 4):**
```javascript
const NOTES_KEY = 'fnd-notes';

function loadNotes() {
  try {
    const saved = localStorage.getItem(NOTES_KEY);
    if (saved) document.getElementById('notes-area').value = saved;
  } catch (e) { /* localStorage unavailable — no-op */ }
}

document.getElementById('notes-area').addEventListener('input', e => {
  try {
    localStorage.setItem(NOTES_KEY, e.target.value);
  } catch (e) { /* localStorage unavailable — no-op */ }
});
```

5. **GSAP KPI animation (per D-11, RESEARCH Pattern 2):**

document.addEventListener('DOMContentLoaded', () => {
  gsap.from('.kpi-tile', { y: 20, opacity: 0, stagger: 0.08, duration: 0.5, ease: 'power2.out' });

  const targets = [
    { id: 'kpi-pilots',   val: 1,      prefix: '',  suffix: '' },
    { id: 'kpi-invoices', val: 23,     prefix: '',  suffix: '' },

    const el = document.getElementById(id);

    gsap.to(obj, {
      v: val,
      duration: 1.2,
      ease: 'power2.out',
      snap: { v: 1 },
      onUpdate: () => {
        el.textContent = prefix + Math.round(obj.v).toLocaleString('en-US') + suffix;
      }
    });
  });

  initTasks();
  loadNotes();
});
